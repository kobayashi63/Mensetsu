\begin{question}
  行列式の定義
  \[
  \vert A\vert =\sum_{\sigma \in S_n} (\mathrm{sgn}\sigma) a_{1\sigma(1)} a_{2\sigma(2)}\cdots a_{n\sigma(n)}
  \]
  を用いて，次の行列式の値を計算せよ．
  \[
  \begin{vmatrix}
    1 & 0 & 1 & 2 \\
    2 & 1 & 2 & 4 \\
    3 & 1 & 1 & 2 \\
    1 & 3 & 2 & 1
  \end{vmatrix} 
  \]
\end{question}
